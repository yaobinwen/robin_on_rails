\section{Alignment}

  0 & \in U_1 \\
  0 & \in U_2 \\
    & \ldots \\
+ \ 0 & \in U_m \\
= \ 0 & \in U_1 + U_2 + \ldots + U_m
\end{align*}

\begin{align*}
  w_1 + w_2 & = (u_{11} + \ldots + u_{m1}) + (u_{12} + \ldots + u_{m2}) \\
            & = u_{11} + \ldots + u_{m1} + u_{12} + \ldots + u_{m2} \\
            & = (u_{11} + u_{12}) + \ldots + (u_{m1} + u_{m2})
\end{align*}
